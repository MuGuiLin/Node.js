Koa-Ejs 【.use(Views('./template'), {map: {html: 'ejs'}})】

配置模板引擎中间件：.use(Views('./template',{extension: 'ejs'}))

        
            const Koa = require('koa');
            const render = require('koa-ejs');
            const path = require('path');
            
            const app = new Koa();
            render(app, {
            root: path.join(__dirname, 'view'),
            layout: 'template',
            viewExt: 'html',
            cache: false,
            debug: true
            });
            
            app.use(async function (ctx) {
            await ctx.render('user');
            });
            
            app.listen(7001);
